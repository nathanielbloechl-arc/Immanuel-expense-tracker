        async function renderRegistry() {

            const records = await db.expenses.orderBy('date').reverse().toArray();

            if (records.length === 0) {

                ledgerBody.innerHTML = '';

                emptyState.classList.remove('hidden');

                totalDisplay.innerText = "$0.00";

                return;

            }

            

            emptyState.classList.add('hidden');

            let htmlBuffer = '';

            let totalRunning = 0;


            records.forEach(item => {

                totalRunning += parseFloat(item.amount);

                let receiptUi = '<span class="text-gray-300 text-xs">—</span>';

                if (item.imageBlob) {

                    receiptUi = `

                        <button onclick="viewReceipt(${item.id})" class="inline-flex items-center text-blue-600 hover:text-blue-800 bg-blue-50 px-2 py-1 rounded text-xs font-medium">

                            View Image

                        </button>

                    `;

                }


                htmlBuffer += `

                    <tr class="hover:bg-gray-50/70 transition-colors">

                        <td class="py-3.5 px-4 font-mono text-xs text-gray-500">${item.date}</td>

                        <td class="py-3.5 px-4">

                            <div class="flex justify-between items-start mb-0.5">

                                <span class="font-semibold text-gray-900">${escapeHtml(item.vendor)}</span>

                                <span class="font-bold text-gray-900 ml-4">$${parseFloat(item.amount).toFixed(2)}</span>

                            </div>

                            <span class="text-xs text-gray-500 block truncate max-w-[280px]">${escapeHtml(item.description || '')}</span>

                            <span class="inline-block bg-blue-50 text-blue-700 text-[10px] px-2 py-0.5 rounded-md font-semibold mt-1">${item.category}</span>

                        </td>

                        <td class="py-3.5 px-4 text-center whitespace-nowrap">${receiptUi}</td>

                        <td class="py-3.5 px-4 text-right">

                            <button onclick="deleteRecord(${item.id})" class="text-gray-400 hover:text-red-600 font-medium text-xs">

                                Delete

                            </button>

                        </td>

                    </tr>

                `;

            });

        }

            renderRegistry();

        async function viewReceipt(id) {

            const entry = await db.expenses.get(id);

            if (entry && entry.imageBlob) {

                modalImg.src = URL.createObjectURL(entry.imageBlob);

                imageModal.classList.remove('hidden');

                setTimeout(() => { imageModal.classList.remove('opacity-0'); }, 10);

            }

        }


        function closeModal() {

            imageModal.classList.add('opacity-0');

            setTimeout(() => {

                imageModal.classList.add('hidden');

                if (modalImg.src.startsWith('blob:')) URL.revokeObjectURL(modalImg.src);

                modalImg.src = '';

            }, 200);

        }


        async function deleteRecord(id) {

            if (confirm("Delete this transaction entry?")) {

                await db.expenses.delete(id);

                renderRegistry();

            }

        }

                        <div class="receipt-block">

                            </div>

            });


            await Promise.all(promises);


            // Execute browser print sheet window

            window.print();


            // Clear binary memory allocation pointers

            attachmentsArray.forEach(r => URL.revokeObjectURL(r.url));

            printContainer.innerHTML = '';
